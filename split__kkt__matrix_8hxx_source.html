275inline bool SplitKKTMatrix::isDimensionConsistent() const {
276 if (Fxx.rows() != 2*dimv_) return false;
277 if (Fxx.cols() != 2*dimv_) return false;
278 if (Fvu.rows() != dimv_) return false;
279 if (Fvu.cols() != dimu_) return false;
280 if (Qxx.rows() != 2*dimv_) return false;
281 if (Qxx.cols() != 2*dimv_) return false;
282 if (Qaa.rows() != dimv_) return false;
283 if (Qaa.cols() != dimv_) return false;
284 if (Qxu.rows() != 2*dimv_) return false;
285 if (Qxu.cols() != dimu_) return false;
286 if (Quu.rows() != dimu_) return false;
287 if (Quu.cols() != dimu_) return false;
288 if (has_floating_base_) {
289 if (Fqq_prev.rows() != dimv_) return false;
290 if (Fqq_prev.cols() != dimv_) return false;
291 }
292 if (fx.size() != 2*dimv_) return false;
293 if (hx.size() != 2*dimv_) return false;
294 if (ha.size() != dimv_) return false;
295 if (hu.size() != dimu_) return false;
296 return true;
297}

300inline bool SplitKKTMatrix::isApprox(const SplitKKTMatrix& other) const {
301 if (!Fxx.isApprox(other.Fxx)) return false;
302 if (!Fvu.isApprox(other.Fvu)) return false;
303 if (!Qxx.isApprox(other.Qxx)) return false;
304 if (!Qaa.isApprox(other.Qaa)) return false;
305 if (!Qxu.isApprox(other.Qxu)) return false;
306 if (!Quu.isApprox(other.Quu)) return false;
307 if (!Qff().isApprox(other.Qff())) return false;
308 if (!Qqf().isApprox(other.Qqf())) return false;
309 if (!Fqq_prev.isApprox(other.Fqq_prev)) return false;
310 if (!fx.isApprox(other.fx)) return false;
311 Eigen::VectorXd vec(2), other_vec(2);
312 vec << Qtt, Qtt_prev;
313 other_vec << other.Qtt, other.Qtt_prev;
314 if (!vec.isApprox(other_vec)) return false;
315 if (!hx.isApprox(other.hx)) return false;
316 if (!hu.isApprox(other.hu)) return false;
317 if (!ha.isApprox(other.ha)) return false;
318 if (!hf().isApprox(other.hf())) return false;
319 return true;
320}

345inline void SplitKKTMatrix::setRandom() {
346 Fxx.setRandom();
347 Fvu.setRandom();
348 const Eigen::MatrixXd Qxxuu_seed = Eigen::MatrixXd::Random(dimx_+dimu_, dimx_+dimu_);
349 const Eigen::MatrixXd Qxxuu = Qxxuu_seed * Qxxuu_seed.transpose();
350 Qxx = Qxxuu.topLeftCorner(dimx_, dimx_);
351 Qxu = Qxxuu.topRightCorner(dimx_, dimu_);
352 Quu = Qxxuu.bottomRightCorner(dimu_, dimu_);
353 const Eigen::MatrixXd Qaaff_seed = Eigen::MatrixXd::Random(dimv_+dimf_, dimv_+dimf_);
354 const Eigen::MatrixXd Qaaff = Qaaff_seed * Qaaff_seed.transpose();
355 Qaa = Qaaff.topLeftCorner(dimv_, dimv_);
356 Qff() = Qaaff.bottomRightCorner(dimf_, dimf_);
357 Qqf().setRandom();
358 Fqq_prev.setRandom();
359 fx.setRandom();
360 Qtt = Eigen::VectorXd::Random(1)[0];
361 Qtt_prev = Eigen::VectorXd::Random(1)[0];
362 hx.setRandom();
363 hu.setRandom();
364 ha.setRandom();
365 hf().setRandom();
366}
