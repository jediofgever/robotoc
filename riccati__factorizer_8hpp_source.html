303 const LQRPolicy& lqr_policy, SplitDirection& d, SplitDirectionType& d_next,
304 const bool sto, const bool has_next_sto_phase) const {
305 d.du.noalias() = lqr_policy.K * d.dx;
306 d.du.noalias() += lqr_policy.k;
307 if (sto) {
308 d.du.noalias() += lqr_policy.T * (d.dts_next-d.dts);
309 if (has_next_sto_phase) {
310 d.du.noalias() -= lqr_policy.W * d.dts_next;
311 }
312 }
313 d_next.dx = kkt_residual.Fx;
314 d_next.dx.noalias() += kkt_matrix.Fxx * d.dx;
315 d_next.dv().noalias() += kkt_matrix.Fvu * d.du;
316 if (sto) {
317 d_next.dx.noalias() += kkt_matrix.fx * (d.dts_next-d.dts);
318 }
319 d_next.dts = d.dts;
320 d_next.dts_next = d.dts_next;
321 }
