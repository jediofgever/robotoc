27inline bool ImpulseStatus::operator==(const ImpulseStatus& other) const {
28 assert(other.maxNumContacts() == maxNumContacts());
29 for (int i=0; i<maxNumContacts(); ++i) {
30 if (other.isImpulseActive(i) != isImpulseActive(i)) {
31 return false;
32 }
33 if (!other.contactPlacement(i).isApprox(contactPlacement(i))) {
34 return false;
35 }
36 }
37 return true;
38}
