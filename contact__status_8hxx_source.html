105inline void ContactStatus::activateContact(const int contact_index) {
106 assert(contact_index >= 0);
107 assert(contact_index < max_num_contacts_);
108 if (!is_contact_active_[contact_index]) {
109 is_contact_active_[contact_index] = true;
110 switch (contact_types_[contact_index]) {
111 case ContactType::PointContact:
112 dimf_ += 3;
113 break;
114 case ContactType::SurfaceContact:
115 dimf_ += 6;
116 break;
117 default:
118 break;
119 }
120 }
121 set_has_active_contacts();
122}
