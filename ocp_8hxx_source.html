12inline OCP::OCP(const Robot& robot, const std::shared_ptr<CostFunction>& cost,

15 const std::shared_ptr<STOConstraints>& sto_constraints,
16 const double T, const int N,
17 const int max_num_each_discrete_events)
18 : data(N, SplitOCP(robot, cost, constraints)),
19 aux(max_num_each_discrete_events, SplitOCP(robot, cost, constraints)),
20 lift(max_num_each_discrete_events, SplitOCP(robot, cost, constraints)),
21 impulse(max_num_each_discrete_events, ImpulseSplitOCP(robot, cost, constraints)),
22 terminal(TerminalOCP(robot, cost, constraints)),
23 robot_(robot),
24 cost_(cost),
25 constraints_(constraints),
26 sto_cost_(sto_cost),
27 sto_constraints_(sto_constraints),

29 T_(T),
30 N_(N),
31 max_num_each_discrete_events_(max_num_each_discrete_events),
32 is_sto_enabled_(true) {
33 try {
34 if (T <= 0) {
35 throw std::out_of_range("invalid value: T must be positive!");
36 }
37 if (N <= 0) {
38 throw std::out_of_range("invalid value: N must be positive!");
39 }
40 if (max_num_each_discrete_events < 0) {
41 throw std::out_of_range(
42 "invalid argument: max_num_each_discrete_events must be non-negative!");
43 }
44 const auto& min_dt = sto_constraints->minimumDwellTimes();
45 double sum_min_dt = 0;
46 for (const auto e : min_dt) {
47 sum_min_dt += e;
48 }
49 if (T <= sum_min_dt) {
50 throw std::out_of_range(
51 "invalid argument: sum of the minimum dwell-times must be smaller than T!");
52 }
53 }
54 catch(const std::exception& e) {
55 std::cerr << e.what() << '\n';
56 std::exit(EXIT_FAILURE);
57 }
58 discretization_.setDiscretizationMethod(DiscretizationMethod::PhaseBased);
59}

118inline void OCP::resize(const int max_num_each_discrete_events) {
119 assert(max_num_each_discrete_events >= 0);
120 assert(impulse.size() == discretization_.maxNumEachDiscreteEvents());
121 assert(aux.size() == discretization_.maxNumEachDiscreteEvents());
122 assert(lift.size() == discretization_.maxNumEachDiscreteEvents());
123 if (max_num_each_discrete_events > max_num_each_discrete_events_) {
124 const auto discretization_method = discretization_.discretizationMethod();
125 discretization_ = TimeDiscretization(T_, N_, max_num_each_discrete_events);
126 discretization_.setDiscretizationMethod(discretization_method);
127 while (max_num_each_discrete_events > impulse.size()) {
128 impulse.emplace_back(robot_, cost_, constraints_);
129 aux.emplace_back(robot_, cost_, constraints_);
130 }
131 while (max_num_each_discrete_events > lift.size()) {
132 lift.emplace_back(robot_, cost_, constraints_);
133 }
134 max_num_each_discrete_events_ = max_num_each_discrete_events;
135 }
136}
