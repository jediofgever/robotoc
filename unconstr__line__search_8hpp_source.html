96 template <typename UnconstrOCPType>
97 double computeStepSize(UnconstrOCPType& ocp, aligned_vector<Robot>& robots,
98 const double t, const Eigen::VectorXd& q,
99 const Eigen::VectorXd& v, const Solution& s,
100 const Direction& d,
101 const double max_primal_step_size) {
102 assert(max_primal_step_size > 0);
103 assert(max_primal_step_size <= 1);
104 // If filter is empty, augment the current solution to the filter.
105 if (filter_.isEmpty()) {
106 computeCostAndViolation(ocp, robots, t, q, v, s);
107 filter_.augment(totalCosts(), totalViolations());
108 }
109 double primal_step_size = max_primal_step_size;
110 while (primal_step_size > min_step_size_) {
111 computeSolutionTrial(s, d, primal_step_size);
112 computeCostAndViolation(ocp, robots, t, q, v, s_trial_, primal_step_size);
113 const double total_costs = totalCosts();
114 const double total_violations = totalViolations();
115 if (filter_.isAccepted(total_costs, total_violations)) {
116 filter_.augment(total_costs, total_violations);
117 break;
118 }
119 primal_step_size *= step_size_reduction_rate_;
120 }
121 if (primal_step_size > min_step_size_) {
122 return primal_step_size;
123 }
124 else {
125 return min_step_size_;
126 }
127 }
