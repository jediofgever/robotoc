120 static void computeSolutionTrial(const Robot& robot, const SplitSolution& s,
121 const SplitDirection& d,
122 const double step_size,
123 SplitSolution& s_trial) {
124 s_trial.setContactStatus(s);
125 robot.integrateConfiguration(s.q, d.dq(), step_size, s_trial.q);
126 s_trial.v = s.v + step_size * d.dv();
127 s_trial.a = s.a + step_size * d.da();
128 s_trial.u = s.u + step_size * d.du;
129 if (s.hasActiveContacts()) {
130 s_trial.f_stack() = s.f_stack() + step_size * d.df();
131 s_trial.set_f_vector();
132 }
133 }
