173 static void linearizeStateEquation_impl(const Robot& robot, const double dt,
174 const Eigen::VectorXd& q_prev,
175 const SplitSolution& s,
176 const SplitSolutionType& s_next,
177 SplitKKTMatrix& kkt_matrix,
178 SplitKKTResidual& kkt_residual) {
179 assert(dt > 0);
180 assert(q_prev.size() == robot.dimq());
181 evalStateEquation(robot, dt, s, s_next.q, s_next.v, kkt_residual);
182 if (robot.hasFloatingBase()) {
183 robot.dSubtractConfiguration_dqf(s.q, s_next.q, kkt_matrix.Fqq());
184 robot.dSubtractConfiguration_dq0(q_prev, s.q, kkt_matrix.Fqq_prev);
185 kkt_residual.lq().template head<6>().noalias()
186 += kkt_matrix.Fqq().template topLeftCorner<6, 6>().transpose()
187 * s_next.lmd.template head<6>();
188 kkt_residual.lq().template head<6>().noalias()
189 += kkt_matrix.Fqq_prev.template topLeftCorner<6, 6>().transpose()
190 * s.lmd.template head<6>();
191 kkt_residual.lq().tail(robot.dimv()-6).noalias()
192 += s_next.lmd.tail(robot.dimv()-6) - s.lmd.tail(robot.dimv()-6);
193 }
194 else {
195 kkt_matrix.Fqq().diagonal().fill(1.);
196 kkt_residual.lq().noalias() += s_next.lmd - s.lmd;
197 }
198 kkt_matrix.Fqv().diagonal().fill(dt);
199 kkt_residual.lv().noalias() += dt * s_next.lmd + s_next.gmm - s.gmm;
200 kkt_residual.la.noalias() += dt * s_next.gmm;
201 // STO sensitivities
202 kkt_residual.h += s_next.lmd.dot(s.v);
203 kkt_residual.h += s_next.gmm.dot(s.a);
204 kkt_matrix.hv().noalias() += s_next.lmd;
205 kkt_matrix.ha.noalias() += s_next.gmm;
206 kkt_matrix.fq() = s.v;
207 kkt_matrix.fv() = s.a;
208 }
