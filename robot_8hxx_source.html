23template <typename TangentVectorType, typename ConfigVectorType>
24inline void Robot::integrateConfiguration(
25 const Eigen::MatrixBase<TangentVectorType>& v,
26 const double integration_length,
27 const Eigen::MatrixBase<ConfigVectorType>& q) const {
28 assert(v.size() == dimv_);
29 assert(q.size() == dimq_);
30 if (has_floating_base_) {
31 const Eigen::VectorXd q_tmp = q;
32 pinocchio::integrate(model_, q_tmp, integration_length*v,
33 const_cast<Eigen::MatrixBase<ConfigVectorType>&>(q));
34 }
35 else {
36 (const_cast<Eigen::MatrixBase<ConfigVectorType>&>(q)).noalias()
37 += integration_length * v;
38 }
39}

240template <typename Vector3dType, typename MatrixType>
241inline void Robot::getJacobianTransformFromLocalToWorld(
242 const int frame_id, const Eigen::MatrixBase<Vector3dType>& vec_world,
243 const Eigen::MatrixBase<MatrixType>& J) {
244 assert(vec_world.size() == 3);
245 assert(J.rows() == 6);
246 assert(J.cols() == dimv_);
247 const_cast<Eigen::MatrixBase<MatrixType>&>(J).setZero();
248 getFrameJacobian(frame_id, const_cast<Eigen::MatrixBase<MatrixType>&>(J));
249 for (int i=0; i<dimv_; ++i) {
250 const_cast<Eigen::MatrixBase<MatrixType>&>(J).template topRows<3>().col(i).noalias()
251 = J.template bottomRows<3>().col(i).cross(vec_world.template head<3>());
252 }

256template <typename VectorType>
257inline void Robot::computeBaumgarteResidual(
258 const ContactStatus& contact_status,
259 const Eigen::MatrixBase<VectorType>& baumgarte_residual) {
260 assert(baumgarte_residual.size() == contact_status.dimf());
261 int dimf_stack = 0;
262 int point_contact_index = 0;
263 int surface_contact_index = 0;
264 for (int i=0; i<contact_frames_.size(); ++i) {
265 switch (contact_types_[i]) {
266 case ContactType::PointContact:
267 if (contact_status.isContactActive(i)) {
268 point_contacts_[point_contact_index].computeBaumgarteResidual(
269 model_, data_, contact_status.contactPosition(i),
270 (const_cast<Eigen::MatrixBase<VectorType>&>(baumgarte_residual))
271 .template segment<3>(dimf_stack));
272 dimf_stack += 3;
273 }
274 point_contact_index += 1;
275 break;
276 case ContactType::SurfaceContact:
277 if (contact_status.isContactActive(i)) {
278 surface_contacts_[surface_contact_index].computeBaumgarteResidual(
279 model_, data_, contact_status.contactPlacement(i),
280 (const_cast<Eigen::MatrixBase<VectorType>&>(baumgarte_residual))
281 .template segment<6>(dimf_stack));
282 dimf_stack += 6;
283 }
284 surface_contact_index += 1;
285 break;
286 default:
287 break;
288 }
289 }
290}

502inline void Robot::setContactForces(const ContactStatus& contact_status,
503 const std::vector<Vector6d>& f) {
504 assert(f.size() == max_num_contacts_);
505 int point_contact_index = 0;
506 int surface_contact_index = 0;
507 for (int i=0; i<contact_types_.size(); ++i) {
508 switch (contact_types_[i]) {
509 case ContactType::PointContact:
510 if (contact_status.isContactActive(i)) {
511 point_contacts_[point_contact_index].computeJointForceFromContactForce(
512 f[i].template head<3>(), fjoint_);
513 }
514 else {
515 point_contacts_[point_contact_index].computeJointForceFromContactForce(
516 Eigen::Vector3d::Zero(), fjoint_);
517 }
518 ++point_contact_index;
519 break;
520 case ContactType::SurfaceContact:
521 if (contact_status.isContactActive(i)) {
522 surface_contacts_[surface_contact_index].computeJointForceFromContactWrench(
523 f[i], fjoint_);
524 }
525 else {
526 surface_contacts_[surface_contact_index].computeJointForceFromContactWrench(
527 Vector6d::Zero(), fjoint_);
528 }
529 ++surface_contact_index;
530 break;
531 default:
532 break;
533 }
534 }

595template <typename ConfigVectorType, typename TangentVectorType1,
596 typename TangentVectorType2, typename MatrixType1,
597 typename MatrixType2, typename MatrixType3>
598inline void Robot::RNEADerivatives(
599 const Eigen::MatrixBase<ConfigVectorType>& q,
600 const Eigen::MatrixBase<TangentVectorType1>& v,
601 const Eigen::MatrixBase<TangentVectorType2>& a,
602 const Eigen::MatrixBase<MatrixType1>& dRNEA_partial_dq,
603 const Eigen::MatrixBase<MatrixType2>& dRNEA_partial_dv,
604 const Eigen::MatrixBase<MatrixType3>& dRNEA_partial_da) {
605 assert(q.size() == dimq_);
606 assert(v.size() == dimv_);
607 assert(a.size() == dimv_);
608 assert(dRNEA_partial_dq.cols() == dimv_);
609 assert(dRNEA_partial_dq.rows() == dimv_);
610 assert(dRNEA_partial_dv.cols() == dimv_);
611 assert(dRNEA_partial_dv.rows() == dimv_);
612 assert(dRNEA_partial_da.cols() == dimv_);
613 assert(dRNEA_partial_da.rows() == dimv_);
614 if (max_num_contacts_) {
615 pinocchio::computeRNEADerivatives(
616 model_, data_, q, v, a, fjoint_,
617 const_cast<Eigen::MatrixBase<MatrixType1>&>(dRNEA_partial_dq),
618 const_cast<Eigen::MatrixBase<MatrixType2>&>(dRNEA_partial_dv),
619 const_cast<Eigen::MatrixBase<MatrixType3>&>(dRNEA_partial_da));
620 }
621 else {
622 pinocchio::computeRNEADerivatives(
623 model_, data_, q, v, a,
624 const_cast<Eigen::MatrixBase<MatrixType1>&>(dRNEA_partial_dq),
625 const_cast<Eigen::MatrixBase<MatrixType2>&>(dRNEA_partial_dv),
626 const_cast<Eigen::MatrixBase<MatrixType3>&>(dRNEA_partial_da));
627 }

689template <typename MatrixType1, typename MatrixType2, typename MatrixType3>
690inline void Robot::computeMJtJinv(
691 const Eigen::MatrixBase<MatrixType1>& M,
692 const Eigen::MatrixBase<MatrixType2>& J,
693 const Eigen::MatrixBase<MatrixType3>& MJtJinv) {
694 assert(M.rows() == dimv_);
695 assert(M.cols() == dimv_);
696 assert(J.rows() <= max_dimf_);
697 assert(J.cols() == dimv_);
698 assert(MJtJinv.rows() == M.rows()+J.rows());
699 assert(MJtJinv.cols() == M.rows()+J.rows());
700 const int dimf = J.rows();
701 data_.M = M;
702 pinocchio::cholesky::decompose(model_, data_);
703 data_.sDUiJt.leftCols(dimf) = J.transpose();
704 pinocchio::cholesky::Uiv(model_, data_, data_.sDUiJt.leftCols(dimf));
705 for (Eigen::DenseIndex k=0; k<dimv_; ++k) {
706 data_.sDUiJt.leftCols(dimf).row(k) /= std::sqrt(data_.D[k]);
707 }
708 data_.JMinvJt.topLeftCorner(dimf, dimf).noalias()
709 = data_.sDUiJt.leftCols(dimf).transpose() * data_.sDUiJt.leftCols(dimf);
710 if (contact_inv_damping_ > 0.) {
711 data_.JMinvJt.diagonal().array() += contact_inv_damping_;
712 }
713 data_.llt_JMinvJt.compute(data_.JMinvJt.topLeftCorner(dimf, dimf));
714 assert(data_.llt_JMinvJt.info() == Eigen::Success);
715 Eigen::Block<MatrixType3> topLeft
716 = const_cast<Eigen::MatrixBase<MatrixType3>&>(MJtJinv).topLeftCorner(dimv_, dimv_);
717 Eigen::Block<MatrixType3> topRight
718 = const_cast<Eigen::MatrixBase<MatrixType3>&>(MJtJinv).topRightCorner(dimv_, dimf);
719 Eigen::Block<MatrixType3> bottomLeft
720 = const_cast<Eigen::MatrixBase<MatrixType3>&>(MJtJinv).bottomLeftCorner(dimf, dimv_);
721 Eigen::Block<MatrixType3> bottomRight
722 = const_cast<Eigen::MatrixBase<MatrixType3>&>(MJtJinv).bottomRightCorner(dimf, dimf);
723 bottomRight = - pinocchio::Data::MatrixXs::Identity(dimf, dimf);
724 topLeft.setIdentity();
725 data_.llt_JMinvJt.solveInPlace(bottomRight);
726 pinocchio::cholesky::solve(model_, data_, topLeft);
727 bottomLeft.noalias() = J * topLeft;
728 topRight.noalias() = bottomLeft.transpose() * (-bottomRight);
729 topLeft.noalias() -= topRight*bottomLeft;
730 bottomLeft = topRight.transpose();
731 assert(!MJtJinv.hasNaN());
732}
