cmake_minimum_required(VERSION 3.1)
project(idocp_examples_anymal CXX)

set(CMAKE_CXX_STANDARD 11)

find_package(idocp REQUIRED)

add_executable(
    anymal_trotting
    anymal_trotting.cpp
)
target_link_libraries(
    anymal_trotting
    PRIVATE
    idocp::idocp
)
target_include_directories(
    anymal_trotting
    PRIVATE
    ${IDOCP_INCLUDE_DIR}
)
if (ENABLE_VIEWER)
    target_compile_definitions(
        anymal_trotting
        PRIVATE
        ENABLE_VIEWER    
    )
endif()
