109 template <typename VectorType1, typename VectorType2>
110 static void frictionConeResidual(const double mu,
111 const Eigen::MatrixBase<VectorType1>& f_world,
112 const Eigen::Matrix3d& R_surface,
113 const Eigen::MatrixBase<VectorType2>& res) {
114 assert(mu > 0);
115 assert(f_world.size() == 3);
116 assert((R_surface*R_surface.transpose()).isIdentity());
117 assert(res.size() == 5);
118 const Eigen::Vector3d f_local = R_surface.transpose() * f_world;
119 const_cast<Eigen::MatrixBase<VectorType2>&>(res).coeffRef(0) = - f_local.coeff(2);
120 const_cast<Eigen::MatrixBase<VectorType2>&>(res).coeffRef(1)
121 = f_local.coeff(0) - mu * f_local.coeff(2) / std::sqrt(2);
122 const_cast<Eigen::MatrixBase<VectorType2>&>(res).coeffRef(2)
123 = - f_local.coeff(0) - mu * f_local.coeff(2) / std::sqrt(2);
124 const_cast<Eigen::MatrixBase<VectorType2>&>(res).coeffRef(3)
125 = f_local.coeff(1) - mu * f_local.coeff(2) / std::sqrt(2);
126 const_cast<Eigen::MatrixBase<VectorType2>&>(res).coeffRef(4)
127 = - f_local.coeff(1) - mu * f_local.coeff(2) / std::sqrt(2);
128 }
