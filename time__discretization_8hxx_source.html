80inline void TimeDiscretization::discretize(
81 const std::shared_ptr<ContactSequence>& contact_sequence, const double t) {
82 if (discretization_method_ == DiscretizationMethod::GridBased) {
83 countDiscreteEvents(contact_sequence, t, true);
84 countTimeStepsGridBased(t);
85 }
86 else {
87 countDiscreteEvents(contact_sequence, t, false);
88 countTimeStepsPhaseBased(t);
89 }
90 countTimeStages();
91 countContactPhase();
92 countSTOEvents();
93 assert(isFormulationTractable());
94 assert(isSwitchingTimeConsistent());
95}

429inline bool TimeDiscretization::isFormulationTractable() const {
430 for (int i=0; i<N(); ++i) {
431 if (isTimeStageBeforeImpulse(i) && isTimeStageBeforeLift(i)) {
432 return false;
433 }
434 }
435 for (int i=0; i<N()-1; ++i) {
436 if (isTimeStageBeforeImpulse(i) && isTimeStageBeforeImpulse(i+1)) {
437 return false;
438 }
439 if (isTimeStageBeforeLift(i) && isTimeStageBeforeImpulse(i+1)) {
440 return false;
441 }
442 }
443 return true;
444}

447inline bool TimeDiscretization::isSwitchingTimeConsistent() const {
448 for (int i=0; i<N_impulse(); ++i) {
449 if (impulseTime(i) < t0()+eps_ || impulseTime(i) >= tf()-eps_) {
450 return false;
451 }
452 }
453 for (int i=0; i<N_lift(); ++i) {
454 if (liftTime(i) < t0()+eps_ || liftTime(i) > tf()-eps_) {
455 return false;
456 }
457 }
458 return true;

462inline void TimeDiscretization::countDiscreteEvents(
463 const std::shared_ptr<ContactSequence>& contact_sequence, const double t,
464 const bool refine_grids) {
465 const int max_num_impulse_events = contact_sequence->numImpulseEvents();
466 assert(max_num_impulse_events <= max_num_each_discrete_events_);
467 const bool is_phase_based
468 = (discretization_method_ == DiscretizationMethod::PhaseBased);
469 N_impulse_ = 0;
470 for (int impulse_index=0; impulse_index<max_num_impulse_events; ++impulse_index) {
471 const double t_impulse = contact_sequence->impulseTime(impulse_index);
472 if (t_impulse >= t+T_-eps_) {
473 break;
474 }
475 grid_impulse_[impulse_index].t = t_impulse;
476 if (refine_grids) {
477 time_stage_before_impulse_[impulse_index] = std::floor((t_impulse-t)/dt_ideal_);
478 }
479 sto_impulse_[impulse_index]
480 = (is_phase_based && contact_sequence->isSTOEnabledImpulse(impulse_index));
481 ++N_impulse_;
482 }
483 const int max_num_lift_events = contact_sequence->numLiftEvents();
484 assert(max_num_lift_events <= max_num_each_discrete_events_);
485 N_lift_ = 0;
486 for (int lift_index=0; lift_index<max_num_lift_events; ++lift_index) {
487 const double t_lift = contact_sequence->liftTime(lift_index);
488 if (t_lift >= t+T_-eps_) {
489 break;
490 }
491 grid_lift_[lift_index].t = t_lift;
492 if (refine_grids) {
493 time_stage_before_lift_[lift_index] = std::floor((t_lift-t)/dt_ideal_);
494 }
495 sto_lift_[lift_index]
496 = (is_phase_based && contact_sequence->isSTOEnabledLift(lift_index));
497 ++N_lift_;
498 }
499 N_ = N_ideal_;
500 for (int i=0; i<N_impulse_+N_lift_; ++i) {
501 event_types_[i] = contact_sequence->eventType(i);
502 }
503 assert(isFormulationTractable());
504}

507inline void TimeDiscretization::countTimeStepsGridBased(const double t) {
508 int impulse_index = 0;
509 int lift_index = 0;
510 int num_events_on_grid = 0;
511 int time_stage_before_prev_event = 0;
512 int phase = 0;
513 for (int i=0; i<N_ideal_; ++i) {
514 const int stage = i - num_events_on_grid;
515 if (i == time_stage_before_impulse_[impulse_index]) {
516 grid_[stage].dt = grid_impulse_[impulse_index].t - i * dt_ideal_ - t;
517 assert(grid_[stage].dt >= -eps_);
518 assert(grid_[stage].dt <= dt_ideal_+eps_);
519 if (grid_[stage].dt <= eps_) {
520 time_stage_before_impulse_[impulse_index] = stage - 1;
521 grid_impulse_[impulse_index].dt = dt_ideal_;
522 grid_[stage].t = t + (i-1) * dt_ideal_;
523 ++num_events_on_grid;
524 ++impulse_index;
525 }
526 else if (grid_[stage].dt >= max_dt_) {
527 time_stage_before_impulse_[impulse_index] = i + 1;
528 grid_[stage].t = t + i * dt_ideal_;
529 }
530 else {
531 time_stage_before_impulse_[impulse_index] = stage;
532 grid_impulse_[impulse_index].dt = dt_ideal_ - grid_[stage].dt;
533 grid_[stage].t = t + i * dt_ideal_;
534 ++impulse_index;
535 }
536 time_stage_before_prev_event = time_stage_before_impulse_[impulse_index];
537 ++phase;
538 }
539 else if (i == time_stage_before_lift_[lift_index]) {
540 grid_[stage].dt = grid_lift_[lift_index].t - i * dt_ideal_ - t;
541 assert(grid_[stage].dt >= -eps_);
542 assert(grid_[stage].dt <= dt_ideal_+eps_);
543 if (grid_[stage].dt <= eps_) {
544 time_stage_before_lift_[lift_index] = stage - 1;
545 grid_lift_[lift_index].dt = dt_ideal_;
546 grid_[stage].t = t + (i-1) * dt_ideal_;
547 ++num_events_on_grid;
548 ++lift_index;
549 }
550 else if (grid_[stage].dt >= max_dt_) {
551 time_stage_before_lift_[lift_index] = i + 1;
552 grid_[stage].t = t + i * dt_ideal_;
553 }
554 else {
555 time_stage_before_lift_[lift_index] = stage;
556 grid_lift_[lift_index].dt = dt_ideal_ - grid_[stage].dt;
557 grid_[stage].t = t + i * dt_ideal_;
558 ++lift_index;
559 }
560 time_stage_before_prev_event = time_stage_before_lift_[lift_index];
561 ++phase;
562 }
563 else {
564 grid_[stage].dt = dt_ideal_;
565 grid_[stage].t = t + i * dt_ideal_;
566 }
567 grid_[stage].grid_count_in_phase = stage - time_stage_before_prev_event;
568 }
569 N_ = N_ideal_ - num_events_on_grid;
570 grid_[N_].t = t + T_;
571 grid_[N_].grid_count_in_phase = N_ - time_stage_before_prev_event;
572 for (int i=0; i<phase+1; ++i) {
573 N_phase_[phase] = 1;
574 }
575}

578inline void TimeDiscretization::countTimeStepsPhaseBased(const double t) {
579 int next_impulse_index = 0;
580 int next_lift_index = 0;
581 int time_stage_before_prev_event = 0;
582 double t_prev_event = t;
583 grid_[0].t = t;
584 for (int phase=0; phase<N_impulse()+N_lift(); ++phase) {
585 const int next_event_index = phase;
586 const auto next_event_type = eventType(next_event_index);
587 assert(next_event_type != DiscreteEventType::None);
588 if (next_event_type == DiscreteEventType::Impulse) {
589 const int time_stage_before_next_event
590 = time_stage_before_impulse_[next_impulse_index];
591 const int num_phase_grids = time_stage_before_next_event
592 - time_stage_before_prev_event + 1;
593 N_phase_[phase] = num_phase_grids;
594 const double dt_phase
595 = (impulseTime(next_impulse_index)-t_prev_event) / num_phase_grids;
596 for (int stage=time_stage_before_prev_event+1;
597 stage<=time_stage_before_next_event; ++stage) {
598 grid_[stage].dt = dt_phase;
599 }
600 grid_[time_stage_before_prev_event+1].t = t_prev_event + dt_phase;
601 for (int stage=time_stage_before_prev_event+2;
602 stage<=time_stage_before_next_event; ++stage) {
603 grid_[stage].t = grid_[stage-1].t + dt_phase;
604 }
605 for (int stage=time_stage_before_prev_event+1;
606 stage<=time_stage_before_next_event; ++stage) {
607 grid_[stage].grid_count_in_phase = stage - time_stage_before_prev_event;
608 }
609 time_stage_before_prev_event = time_stage_before_next_event;
610 t_prev_event = impulseTime(next_impulse_index);
611 ++next_impulse_index;
612 }
613 else {
614 const int time_stage_before_next_event
615 = time_stage_before_lift_[next_lift_index];
616 const int num_phase_grids = time_stage_before_next_event
617 - time_stage_before_prev_event + 1;
618 N_phase_[phase] = num_phase_grids;
619 const double dt_phase
620 = (liftTime(next_lift_index)-t_prev_event) / num_phase_grids;
621 for (int stage=time_stage_before_prev_event+1;
622 stage<=time_stage_before_next_event; ++stage) {
623 grid_[stage].dt = dt_phase;
624 }
625 grid_[time_stage_before_prev_event+1].t = t_prev_event + dt_phase;
626 for (int stage=time_stage_before_prev_event+2;
627 stage<=time_stage_before_next_event; ++stage) {
628 grid_[stage].t = grid_[stage-1].t + dt_phase;
629 }
630 for (int stage=time_stage_before_prev_event+1;
631 stage<=time_stage_before_next_event; ++stage) {
632 grid_[stage].grid_count_in_phase = stage - time_stage_before_prev_event;
633 }
634 time_stage_before_prev_event = time_stage_before_next_event;
635 t_prev_event = liftTime(next_lift_index);
636 ++next_lift_index;
637 }
638 }
639 const int last_phase = N_impulse() + N_lift();
640 const int num_phase_grids = N_ - time_stage_before_prev_event;
641 N_phase_[last_phase] = num_phase_grids;
642 const double dt_phase = (t+T_-t_prev_event) / num_phase_grids;
643 for (int stage=time_stage_before_prev_event+1; stage<N_; ++stage) {
644 grid_[stage].dt = dt_phase;
645 }
646 grid_[time_stage_before_prev_event+1].t = t_prev_event + dt_phase;
647 for (int stage=time_stage_before_prev_event+2; stage<=N_; ++stage) {
648 grid_[stage].t = grid_[stage-1].t + dt_phase;
649 }
650 for (int stage=time_stage_before_prev_event+1; stage<=N_; ++stage) {
651 grid_[stage].grid_count_in_phase = stage - time_stage_before_prev_event;
652 }
653 for (int impulse_index=0; impulse_index<N_impulse(); ++impulse_index) {
654 grid_impulse_[impulse_index].dt
655 = grid_[time_stage_before_impulse_[impulse_index]+1].dt;
656 }
657 for (int lift_index=0; lift_index<N_lift(); ++lift_index) {
658 grid_lift_[lift_index].dt
659 = grid_[time_stage_before_lift_[lift_index]+1].dt;
660 }
661 grid_[0].dt = grid_[1].dt;
662 if (N_impulse() > 0) {
663 if (time_stage_before_impulse_[0] == 0) {
664 grid_[0].dt = grid_impulse_[0].t - t;
665 }
666 }
667 if (N_lift() > 0) {
668 if (time_stage_before_lift_[0] == 0) {
669 grid_[0].dt = grid_lift_[0].t - t;
670 }
671 }
672}

675inline void TimeDiscretization::countTimeStages() {
676 int impulse_index = 0;
677 int lift_index = 0;
678 for (int i=0; i<N(); ++i) {
679 if (impulse_index < N_impulse()) {
680 if (i == timeStageBeforeImpulse(impulse_index)) {
681 is_time_stage_before_impulse_[i] = true;
682 impulse_index_after_time_stage_[i] = impulse_index;
683 ++impulse_index;
684 }
685 else {

715inline void TimeDiscretization::countContactPhase() {
716 int num_events = 0;
717 for (int i=0; i<N(); ++i) {
718 grid_[i].contact_phase = num_events;
719 grid_[i].N_phase = N_phase(grid_[i].contact_phase);
720 if (isTimeStageBeforeImpulse(i) || isTimeStageBeforeLift(i)) {
721 ++num_events;
722 }
723 }
724 grid_[N()].contact_phase = num_events;
725 grid_[N()].N_phase = N_phase(grid_[N()].contact_phase);
726 for (int impulse_index=0; impulse_index<N_impulse(); ++impulse_index) {
727 grid_impulse_[impulse_index].contact_phase = contactPhase(timeStageAfterImpulse(impulse_index));
728 grid_impulse_[impulse_index].N_phase = N_phase(grid_impulse_[impulse_index].contact_phase);
729 }
730 for (int lift_index=0; lift_index<N_lift(); ++lift_index) {
731 grid_lift_[lift_index].contact_phase = contactPhase(timeStageAfterLift(lift_index));
732 grid_lift_[lift_index].N_phase = N_phase(grid_lift_[lift_index].contact_phase);
733 }
734}

737inline void TimeDiscretization::countSTOEvents() {
738 const bool is_phase_based
739 = (discretization_method_ == DiscretizationMethod::PhaseBased);
740 int event_index = 0;
741 for (int i=0; i<N(); ++i) {
742 if (isTimeStageBeforeImpulse(i)) {
743 sto_event_[event_index]
744 = (is_phase_based && isSTOEnabledImpulse(impulseIndexAfterTimeStage(i)));
745 ++event_index;
746 }
747 else if (isTimeStageBeforeLift(i)) {
748 sto_event_[event_index]
749 = (is_phase_based && isSTOEnabledLift(liftIndexAfterTimeStage(i)));
750 ++event_index;
751 }
752 }
753 for (int i=N_lift()+N_impulse(); i<sto_event_.size(); ++i) {
754 sto_event_[i] = false;
755 }
756}
